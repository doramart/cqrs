Example
class User extends AbstractActor{

    constructor(name,age){
        this.name = name;
        this.age = age;
    }

    // domain method
    // here don't change data, please use when method.
    change(name,age){
         // don't write `this.name = name`
         this.apply("change",{name,age});
    }

    // change user's data.
    when(event){
         switch(event.name){
             case "change":
               this.name = event.data.name;
               this.age = event.data.age;
             break;
         }
    }

}
